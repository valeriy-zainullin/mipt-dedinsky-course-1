(64)' = 0.
(x)' = 1.
(sin(x))' = cos(x).
(64*sin(x))' = 64*cos(x).
(x)' = 1.
(cos(x))' = -1*sin(x).
(64*sin(x)*cos(x))' = 64*cos(x)*cos(x)+64*sin(x)*-1*sin(x).
(2+2*4*8*sin(x)*cos(x))' = 64*cos(x)*cos(x)+64*sin(x)*-1*sin(x).
